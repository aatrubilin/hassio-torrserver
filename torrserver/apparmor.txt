#include <tunables/global>

profile torrserver flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/bash>
  #include <abstractions/ssl_certs>

  # --- [ PARENT PROCESS PERMISSIONS (S6 + Bashio + run.sh) ] ---

  # CRITICAL: Allow S6-Overlay entrypoint
  /init rix,

  # Allow standard system utilities.
  # Restricting this too much causes crashes because Bashio/S6 use many tools (grep, cat, rm, date, etc.)
  # Security is handled by isolating the child process below.
  /bin/** ix,
  /usr/bin/** ix,
  /sbin/** ix,
  /usr/sbin/** ix,

  # Bashio specific library access
  /usr/lib/bashio/** ix,

  # S6-Overlay service definitions
  /run/s6/** ix,
  /etc/services.d/** rix,
  /etc/cont-init.d/** rix,
  /etc/s6-overlay/** rix,

  # Home Assistant persistent storage paths
  /data/** rw,
  /config/** rwk,      # TorrServer configs and accs.db
  /share/** rw,        # Shared folder for media/torrents

  # Temporary files and system runtime
  /tmp/** rwk,
  /run/{,**} rwk,
  /dev/tty rw,

  # --- [ PROFILE TRANSITION ] ---

  # Transition to a dedicated restricted profile when executing the binary
  /usr/bin/torrserver cx -> torrserver_bin,

  # --- [ CHILD PROCESS (The actual Server) ] ---
  profile torrserver_bin flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>
    #include <abstractions/ssl_certs>

    # Network: Full access for BitTorrent protocol (DHT/PEX) and WebUI
    network,

    # Signals: Allow receiving signals from the parent shell/S6 (e.g., stop/kill) signal (receive) peer=torrserver,

    # Data access
    /config/** rwk,
    /share/** rw,
    /data/** rw,
    /tmp/** rwk,

    # Allow the binary to read itself (required for execution)
    /usr/bin/torrserver r,

    # Go runtime specific requirements for memory management and monitoring
    /proc/self/status r,
    /sys/kernel/mm/transparent_hugepage/hpage_pmd_size r,
    /proc/sys/kernel/mm/transparent_hugepage/hpage_pmd_size r,

    # Standard output and logging
    /dev/stdout rw,
    /dev/stderr rw,
    /dev/tty rw,
  }
}
